import numpy as np
from qiskit import QuantumCircuit, ClassicalRegister, QuantumRegister
from qiskit import execute

q= QuantumRegister(3, 'q')
C= QuantumCircuit(q)
C.x(q[0])
C.x(q[1])
C.cx(q[0], q[1])

from qiskit.tools.visualization import circuit_drawer

circuit_drawer(C)
